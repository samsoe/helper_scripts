https://cloud.google.com/bigquery/docs/loading-data-cloud-storage-csv

*** template ***
bq --location=location load \
--source_format=format \
dataset.table \
path_to_source \
schema

*** example ***
bq load \
    --source_format=CSV \
    --skip_leading_rows=2
    mydataset.mytable \
    gs://mybucket/mydata.csv \
    ./myschema.json

bq load \
    --autodetect \
    --source_format=CSV \
    mydataset.mytable \
    gs://mybucket/mydata*.csv

*** production ***
bq load \
  --autodetect \
  --source_format=CSV \
  --null_marker="NA" \
